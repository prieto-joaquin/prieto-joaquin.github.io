<script>
(function(){
  var tabs=[].slice.call(document.querySelectorAll('.research-tabs button'));
  var panels=[].slice.call(document.querySelectorAll('.research-panel'));
  var names=tabs.map(function(t){return t.getAttribute('data-tab');});
  function activate(name){
    if(names.indexOf(name)===-1){name='published';}
    tabs.forEach(function(t){t.setAttribute('aria-selected', t.getAttribute('data-tab')===name?'true':'false');});
    panels.forEach(function(p){p.hidden=(p.getAttribute('data-panel')!==name);});
  }
  tabs.forEach(function(t){
    t.addEventListener('click',function(){
      var name=t.getAttribute('data-tab');
      activate(name);
      if(window.history&&history.replaceState){history.replaceState(null,'','#'+name);}
    });
  });
  activate((window.location.hash||'').replace('#',''));
})();
</script>
